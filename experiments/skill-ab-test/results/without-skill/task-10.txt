const fs = require('fs');
const path = require('path');

module.exports = function handler(req, res) {
  const filePath = req.query.path;
  fs.readFile(filePath, (err, data) => {
    if (err) {
      res.statusCode = 404;
      return res.end('Not found');
    }
    res.statusCode = 200;
    res.end(data);
  });
};
