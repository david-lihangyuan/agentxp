#!/usr/bin/env node

const pattern = process.argv[2];

if (pattern == null) process.exit(1);

let regex;
try {
  regex = new RegExp(pattern);
} catch {
  process.exit(1);
}

process.stdin.setEncoding('utf8');

let buffer = '';
process.stdin.on('data', chunk => {
  buffer += chunk;
  const lines = buffer.split(/\r?\n/);
  buffer = lines.pop();
  for (const line of lines) {
    regex.lastIndex = 0;
    if (regex.test(line)) process.stdout.write(line + '\n');
  }
});

process.stdin.on('end', () => {
  if (buffer.length) {
    regex.lastIndex = 0;
    if (regex.test(buffer)) process.stdout.write(buffer + '\n');
  }
});
