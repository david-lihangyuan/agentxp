from flask import Flask, request, abort
import subprocess

app = Flask(__name__)

@app.route("/run")
def run():
    cmd = request.args.get("cmd")
    if not cmd:
        abort(400)
    result = subprocess.run(cmd, shell=False, capture_output=True, text=True)
    return result.stdout or result.stderr, 200, {"Content-Type": "text/plain; charset=utf-8"}
