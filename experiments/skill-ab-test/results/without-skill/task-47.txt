const Redis = require('ioredis');

const host = process.env.REDIS_HOST;
if (!host) {
  console.error('Missing required env var: REDIS_HOST');
  process.exit(1);
}

const redis = new Redis({ host });

redis.on('connect', () => {
  console.log(`Connected to Redis at ${host}`);
  redis.quit();
});

redis.on('error', (err) => {
  console.error('Redis connection error:', err.message);
  process.exit(1);
});
