#!/bin/sh

set -eu

if [ "$#" -ne 1 ]; then
  echo "Usage: $0 <cron-job-id>" >&2
  exit 1
fi

JOB_ID=$1

if [ -z "$JOB_ID" ]; then
  echo "Error: cron job ID must not be empty" >&2
  exit 1
fi

if ! command -v crontab >/dev/null 2>&1; then
  echo "Error: crontab command not found" >&2
  exit 1
fi

TMP_FILE=$(mktemp "${TMPDIR:-/tmp}/crontab.XXXXXX") || {
  echo "Error: failed to create temp file" >&2
  exit 1
}

cleanup() {
  rm -f "$TMP_FILE"
}
trap cleanup EXIT INT TERM HUP

if ! crontab -l >"$TMP_FILE" 2>/dev/null; then
  : >"$TMP_FILE"
fi

if ! grep -Fq -- "$JOB_ID" "$TMP_FILE"; then
  echo "Error: cron job ID not found: $JOB_ID" >&2
  exit 1
fi

FILTERED_FILE=$(mktemp "${TMPDIR:-/tmp}/crontab.XXXXXX") || {
  echo "Error: failed to create temp file" >&2
  exit 1
}

cleanup() {
  rm -f "$TMP_FILE" "$FILTERED_FILE"
}
trap cleanup EXIT INT TERM HUP

grep -Fv -- "$JOB_ID" "$TMP_FILE" >"$FILTERED_FILE" || true

if [ ! -s "$FILTERED_FILE" ]; then
  crontab -r
else
  crontab "$FILTERED_FILE"
fi
