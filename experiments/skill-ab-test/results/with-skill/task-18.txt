const fs = require('fs');
const path = require('path');

function fail(message) {
  console.error(message);
  process.exit(1);
}

const configPathArg = process.argv[2];
if (!configPathArg) {
  fail(`Usage: node ${path.basename(process.argv[1])} <config-file>`);
}

const configPath = path.resolve(configPathArg);

let stats;
try {
  stats = fs.statSync(configPath);
} catch (err) {
  fail(`Config file not found or inaccessible: ${configPath}`);
}

if (!stats.isFile()) {
  fail(`Path is not a file: ${configPath}`);
}

let raw;
try {
  raw = fs.readFileSync(configPath, 'utf8');
} catch (err) {
  fail(`Failed to read config file: ${err.message}`);
}

let config;
try {
  config = JSON.parse(raw);
} catch (err) {
  fail(`Invalid JSON in config file: ${err.message}`);
}

const host = config?.database?.primary?.connection?.host;
if (host === undefined || host === null) {
  fail('Missing config.database.primary.connection.host');
}

console.log(host);
