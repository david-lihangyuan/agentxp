def normalize(val, min_val, max_val):
    if max_val == min_val:
        raise ValueError("max_val and min_val must be different")
    return (val - min_val) / (max_val - min_val)


if __name__ == "__main__":
    print(normalize(5, 0, 10))
    try:
        print(normalize(5, 3, 3))
    except ValueError as e:
        print(f"Error: {e}")
