#!/usr/bin/env node

'use strict';

function readStdin() {
  return new Promise((resolve, reject) => {
    const chunks = [];
    process.stdin.on('data', chunk => chunks.push(chunk));
    process.stdin.on('end', () => resolve(Buffer.concat(chunks).toString('utf8')));
    process.stdin.on('error', reject);
  });
}

function extractTraceId(input) {
  const trimmed = input.trim();
  if (!trimmed) {
    throw new Error('No input provided');
  }

  let entry;
  try {
    entry = JSON.parse(trimmed);
  } catch (err) {
    throw new Error(`Invalid JSON input: ${err.message}`);
  }

  const traceId = entry?.event?.detail?.metadata?.traceId;
  if (traceId === undefined || traceId === null) {
    throw new Error('traceId not found at event.detail.metadata.traceId');
  }

  if (typeof traceId === 'object') {
    throw new Error('traceId is not a scalar value');
  }

  return String(traceId);
}

(async () => {
  try {
    const input = await readStdin();
    const traceId = extractTraceId(input);
    process.stdout.write(`${traceId}\n`);
  } catch (err) {
    process.stderr.write(`Error: ${err.message}\n`);
    process.exit(1);
  }
})();
