```sh
#!/bin/sh

set -eu

if [ "$#" -ne 1 ]; then
  echo "Usage: $0 <file-path>" >&2
  exit 1
fi

src=$1
dest_dir=/tmp/archive

if [ ! -e "$src" ]; then
  echo "Error: path does not exist: $src" >&2
  exit 1
fi

if [ ! -f "$src" ]; then
  echo "Error: not a regular file: $src" >&2
  exit 1
fi

if [ ! -r "$src" ]; then
  echo "Error: file is not readable: $src" >&2
  exit 1
fi

if [ ! -d "$dest_dir" ]; then
  if ! mkdir -p "$dest_dir"; then
    echo "Error: failed to create destination directory: $dest_dir" >&2
    exit 1
  fi
fi

if [ ! -w "$dest_dir" ]; then
  echo "Error: destination directory is not writable: $dest_dir" >&2
  exit 1
fi

base=$(basename -- "$src")
dest="$dest_dir/$base"

if [ -e "$dest" ]; then
  echo "Error: destination file already exists: $dest" >&2
  exit 1
fi

if ! mv -- "$src" "$dest"; then
  echo "Error: failed to move file to $dest" >&2
  exit 1
fi
```
