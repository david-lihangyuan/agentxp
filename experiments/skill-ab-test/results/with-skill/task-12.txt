#!/usr/bin/env node

const readline = require('readline');

const pattern = process.argv[2];

if (typeof pattern !== 'string' || pattern.length === 0) {
  process.stderr.write(`Usage: ${process.argv[1]} <regex>\n`);
  process.exit(1);
}

let regex;
try {
  regex = new RegExp(pattern);
} catch (err) {
  process.stderr.write(`Invalid regex: ${err.message}\n`);
  process.exit(1);
}

const rl = readline.createInterface({
  input: process.stdin,
  crlfDelay: Infinity,
});

rl.on('line', (line) => {
  if (regex.test(line)) {
    process.stdout.write(line + '\n');
  }
});

rl.on('close', () => {
  process.exit(0);
});

rl.on('error', (err) => {
  process.stderr.write(`stdin error: ${err.message}\n`);
  process.exit(1);
});
