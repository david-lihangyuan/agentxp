function computePricePerUnit(total_cost, quantity) {
  if (typeof total_cost !== 'number' || Number.isNaN(total_cost)) {
    throw new TypeError('total_cost must be a valid number');
  }
  if (typeof quantity !== 'number' || Number.isNaN(quantity)) {
    throw new TypeError('quantity must be a valid number');
  }
  if (quantity === 0) {
    throw new Error('Division by zero: quantity must not be 0');
  }
  return total_cost / quantity;
}

try {
  console.log(computePricePerUnit(100, 0));
} catch (err) {
  console.error(err.message);
}
