from flask import Flask, request, abort, jsonify
import subprocess

app = Flask(__name__)

ALLOWED_COMMANDS = {
    "uptime": ["uptime"],
    "date": ["date"],
    "whoami": ["whoami"],
}

@app.route("/run", methods=["GET"])
def run_command():
    cmd = request.args.get("cmd")
    if not cmd:
        abort(400, description="Missing cmd parameter")
    if cmd not in ALLOWED_COMMANDS:
        abort(400, description="Unsupported command")

    try:
        result = subprocess.run(
            ALLOWED_COMMANDS[cmd],
            capture_output=True,
            text=True,
            check=False,
            timeout=5,
        )
    except subprocess.TimeoutExpired:
        abort(504, description="Command timed out")
    except Exception:
        abort(500, description="Command execution failed")

    return jsonify(
        command=cmd,
        returncode=result.returncode,
        stdout=result.stdout,
        stderr=result.stderr,
    )
