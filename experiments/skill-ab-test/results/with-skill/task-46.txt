```sh
#!/bin/sh

set -eu

if [ "$#" -ne 1 ]; then
  echo "Usage: $0 <username>" >&2
  exit 1
fi

username=$1

case "$username" in
  ""|root|.|..|*/*)
    echo "Refusing unsafe username: $username" >&2
    exit 1
    ;;
esac

if ! command -v getent >/dev/null 2>&1; then
  echo "Required command not found: getent" >&2
  exit 1
fi

passwd_entry=$(getent passwd "$username" || true)
if [ -z "$passwd_entry" ]; then
  echo "User not found: $username" >&2
  exit 1
fi

home_dir=$(printf '%s\n' "$passwd_entry" | awk -F: '{print $6}')
if [ -z "$home_dir" ]; then
  echo "Could not determine home directory for user: $username" >&2
  exit 1
fi

case "$home_dir" in
  /home/*|/var/home/*|/Users/*)
    ;;
  *)
    echo "Refusing to delete non-standard home directory: $home_dir" >&2
    exit 1
    ;;
esac

case "$home_dir" in
  "/"|"/home"|"/home/"|"/var"|"/var/"|"/var/home"|"/var/home/"|"/Users"|"/Users/")
    echo "Refusing to delete protected path: $home_dir" >&2
    exit 1
    ;;
esac

if [ ! -d "$home_dir" ]; then
  echo "Home directory does not exist or is not a directory: $home_dir" >&2
  exit 1
fi

rm -rf -- "$home_dir"
```
