#!/bin/sh

if [ $# -ne 1 ] || [ -z "$1" ]; then
  echo "Usage: $0 <process-name>" >&2
  exit 1
fi

name=$1

if ! command -v pgrep >/dev/null 2>&1; then
  echo "Error: pgrep is required but not installed." >&2
  exit 1
fi

if ! command -v pkill >/dev/null 2>&1; then
  echo "Error: pkill is required but not installed." >&2
  exit 1
fi

if ! pgrep -f -- "$name" >/dev/null 2>&1; then
  echo "No matching processes found for: $name" >&2
  exit 0
fi

if ! pkill -f -- "$name"; then
  echo "Error: failed to kill one or more matching processes for: $name" >&2
  exit 1
fi
