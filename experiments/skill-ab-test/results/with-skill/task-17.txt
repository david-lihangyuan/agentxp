import os
import sys

try:
    import yaml
except ImportError:
    print("Error: PyYAML is not installed. Install it with: pip install pyyaml", file=sys.stderr)
    sys.exit(1)

settings_path = "settings.yaml"

if not os.path.exists(settings_path):
    print(f"Error: File not found: {settings_path}", file=sys.stderr)
    sys.exit(1)

if not os.path.isfile(settings_path):
    print(f"Error: Not a file: {settings_path}", file=sys.stderr)
    sys.exit(1)

try:
    with open(settings_path, "r", encoding="utf-8") as f:
        data = yaml.safe_load(f)
except PermissionError:
    print(f"Error: Permission denied reading {settings_path}", file=sys.stderr)
    sys.exit(1)
except OSError as e:
    print(f"Error: Failed to read {settings_path}: {e}", file=sys.stderr)
    sys.exit(1)
except yaml.YAMLError as e:
    print(f"Error: Failed to parse YAML in {settings_path}: {e}", file=sys.stderr)
    sys.exit(1)

print(data)
