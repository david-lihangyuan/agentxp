#!/usr/bin/env node

'use strict';

const pattern = process.argv[2];

if (typeof pattern !== 'string') {
  process.stderr.write('Usage: filter-lines <regex-pattern>\n');
  process.exit(1);
}

let regex;
try {
  regex = new RegExp(pattern);
} catch (err) {
  process.stderr.write(`Invalid regex: ${err && err.message ? err.message : 'unknown error'}\n`);
  process.exit(1);
}

process.stdin.setEncoding('utf8');

let buffer = '';

process.stdin.on('data', (chunk) => {
  if (typeof chunk !== 'string') return;
  buffer += chunk;

  const lines = buffer.split(/\r?\n/);
  buffer = lines.pop() ?? '';

  for (const line of lines) {
    if (regex.test(line)) {
      process.stdout.write(line + '\n');
    }
    if (regex.global || regex.sticky) {
      regex.lastIndex = 0;
    }
  }
});

process.stdin.on('end', () => {
  if (buffer.length > 0) {
    if (regex.test(buffer)) {
      process.stdout.write(buffer + '\n');
    }
    if (regex.global || regex.sticky) {
      regex.lastIndex = 0;
    }
  }
});

process.stdin.on('error', (err) => {
  process.stderr.write(`stdin error: ${err && err.message ? err.message : 'unknown error'}\n`);
  process.exit(1);
});
