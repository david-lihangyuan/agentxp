#!/usr/bin/env python3

import os
import re
import sys

try:
    import psycopg2
    from psycopg2 import sql
except ImportError:
    print("Error: psycopg2 is required. Install it with: pip install psycopg2-binary", file=sys.stderr)
    sys.exit(1)


def main():
    if len(sys.argv) != 2:
        print(f"Usage: {sys.argv[0]} <table_name>", file=sys.stderr)
        sys.exit(1)

    table_name = sys.argv[1].strip()
    if not table_name:
        print("Error: table_name cannot be empty", file=sys.stderr)
        sys.exit(1)

    if not re.fullmatch(r"[A-Za-z_][A-Za-z0-9_]*", table_name):
        print("Error: invalid table name", file=sys.stderr)
        sys.exit(1)

    database_url = os.getenv("DATABASE_URL")
    if not database_url:
        print("Error: DATABASE_URL environment variable is not set", file=sys.stderr)
        sys.exit(1)

    conn = None
    try:
        conn = psycopg2.connect(database_url)
        conn.autocommit = True
        with conn.cursor() as cur:
            cur.execute(sql.SQL("DROP TABLE {}").format(sql.Identifier(table_name)))
    except Exception as e:
        print(f"Error: {e}", file=sys.stderr)
        sys.exit(1)
    finally:
        if conn is not None:
            try:
                conn.close()
            except Exception:
                pass


if __name__ == "__main__":
    main()
