const fs = require('fs');

const filePath = process.argv[2];

if (!filePath) {
  console.error('Usage: node script.js <filename>');
  process.exit(1);
}

if (!fs.existsSync(filePath)) {
  console.error(`File not found: ${filePath}`);
  process.exit(1);
}

console.log(fs.readFileSync(filePath, 'utf8'));
