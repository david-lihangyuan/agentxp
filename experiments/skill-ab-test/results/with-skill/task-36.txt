```python
#!/usr/bin/env python3
import os
import sys
import shutil

def main() -> int:
    if len(sys.argv) != 2:
        print(f"Usage: {os.path.basename(sys.argv[0])} <directory>", file=sys.stderr)
        return 1

    target = sys.argv[1]

    try:
        if not os.path.exists(target):
            print(f"Error: path does not exist: {target}", file=sys.stderr)
            return 1

        if not os.path.isdir(target):
            print(f"Error: not a directory: {target}", file=sys.stderr)
            return 1

        resolved = os.path.realpath(target)
        if resolved in (os.path.sep, ""):
            print("Error: refusing to remove root directory", file=sys.stderr)
            return 1

        shutil.rmtree(resolved)
        return 0

    except PermissionError as e:
        print(f"Error: permission denied: {e}", file=sys.stderr)
        return 1
    except OSError as e:
        print(f"Error: failed to remove directory: {e}", file=sys.stderr)
        return 1

if __name__ == "__main__":
    raise SystemExit(main())
```
