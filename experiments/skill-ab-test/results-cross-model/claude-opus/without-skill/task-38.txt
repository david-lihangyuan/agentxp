```javascript
const http = require('http');

const port = process.argv[2];

const server = http.createServer((req, res) => {
  res.statusCode = 200;
  res.setHeader('Content-Type', 'text/plain');
  res.end('Server running\n');
});

server.listen(port, () => {
  console.log(`Server running on port ${port}`);
});
```
