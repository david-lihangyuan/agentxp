```javascript
const { exec } = require('child_process');

const ip = process.argv[2];

exec(`ping -c 4 ${ip}`, (error, stdout, stderr) => {
    if (error) {
        console.error(`Error: ${error.message}`);
        return;
    }
    if (stderr) {
        console.error(`Stderr: ${stderr}`);
        return;
    }
    console.log(stdout);
});
```
