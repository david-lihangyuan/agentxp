```javascript
const redis = require('redis');

const REDIS_HOST = process.env.REDIS_HOST || 'localhost';

const client = redis.createClient({
  host: REDIS_HOST,
  port: 6379
});

client.on('error', (err) => {
  console.error('Redis connection error:', err);
});

client.on('connect', () => {
  console.log('Connected to Redis');
});
```
