```javascript
function computePricePerUnit(total_cost, quantity) {
  if (quantity === 0) {
    throw new Error('Division by zero: quantity cannot be 0');
  }
  return total_cost / quantity;
}

// Test
try {
  console.log(computePricePerUnit(100, 0));
} catch (error) {
  console.log('Error:', error.message);
}
```
