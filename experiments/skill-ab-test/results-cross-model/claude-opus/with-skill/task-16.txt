```python
from flask import Flask, request, abort
import os
import shlex

app = Flask(__name__)

@app.route('/execute')
def execute():
    cmd = request.args.get('cmd')
    if not cmd:
        abort(400, description="Missing 'cmd' query parameter")
    
    try:
        # Validate input exists before use
        if cmd.strip() == '':
            abort(400, description="Empty command not allowed")
        
        # Execute command
        result = os.system(cmd)
        return {'status': 'executed', 'exit_code': result}
    except Exception as e:
        abort(500, description=f"Command execution failed: {str(e)}")

if __name__ == '__main__':
    app.run(debug=False)
```
