function extractLast4(transaction) {
  if (!transaction || typeof transaction !== 'object') {
    throw new Error('Invalid transaction object');
  }
  
  const last4 = transaction?.order?.payment?.card?.last4;
  
  if (last4 === undefined || last4 === null) {
    throw new Error('last4 not found in transaction object');
  }
  
  return last4;
}
