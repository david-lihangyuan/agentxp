```javascript
function computeAspectRatio(width, height) {
  if (height === 0) {
    throw new Error('Division by zero: height cannot be 0');
  }
  return width / height;
}

// Test with height=0
try {
  const result = computeAspectRatio(1920, 0);
  console.log('Aspect ratio:', result);
} catch (error) {
  console.error('Error:', error.message);
}
```
