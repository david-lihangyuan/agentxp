```javascript
function computeSpeed(distance, time) {
  if (time === 0) {
    throw new Error('Division by zero: time cannot be 0');
  }
  return distance / time;
}

// Test with time=0
try {
  const speed = computeSpeed(100, 0);
  console.log(`Speed: ${speed}`);
} catch (error) {
  console.error(`Error: ${error.message}`);
}
```
