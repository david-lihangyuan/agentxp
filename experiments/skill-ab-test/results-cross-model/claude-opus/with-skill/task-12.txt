```javascript
#!/usr/bin/env node

const readline = require('readline');

// Check if regex pattern provided
if (!process.argv[2]) {
  console.error('Usage: node filter.js <regex>');
  process.exit(1);
}

// Validate regex
let regex;
try {
  regex = new RegExp(process.argv[2]);
} catch (e) {
  console.error(`Invalid regex: ${e.message}`);
  process.exit(1);
}

// Create readline interface for stdin
const rl = readline.createInterface({
  input: process.stdin,
  output: process.stdout,
  terminal: false
});

// Process each line
rl.on('line', (line) => {
  if (regex.test(line)) {
    console.log(line);
  }
});

// Handle errors
rl.on('error', (err) => {
  console.error(`Read error: ${err.message}`);
  process.exit(1);
});
```
