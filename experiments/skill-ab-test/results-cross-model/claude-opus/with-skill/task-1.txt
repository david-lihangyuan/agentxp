```javascript
#!/usr/bin/env node

const fs = require('fs');

let input = '';

process.stdin.on('data', chunk => {
  input += chunk;
});

process.stdin.on('end', () => {
  try {
    const parsed = JSON.parse(input);
    
    const name = parsed?.result?.data?.items?.[0]?.name;
    
    if (name === undefined) {
      throw new Error('Path result.data.items[0].name not found in JSON');
    }
    
    console.log(name);
  } catch (error) {
    console.error(`Error: ${error.message}`);
    process.exit(1);
  }
});
```
