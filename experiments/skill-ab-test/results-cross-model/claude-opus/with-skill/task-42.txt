```javascript
#!/usr/bin/env node

const fs = require('fs');
const path = require('path');

// Get input file from command line or stdin
const args = process.argv.slice(2);

function extractTraceId(logEntry) {
    try {
        const parsed = typeof logEntry === 'string' ? JSON.parse(logEntry) : logEntry;
        return parsed?.event?.detail?.metadata?.traceId || null;
    } catch (e) {
        return null;
    }
}

if (args.length > 0) {
    // Read from file
    const filePath = path.resolve(args[0]);
    
    if (!fs.existsSync(filePath)) {
        console.error(`File not found: ${filePath}`);
        process.exit(1);
    }
    
    try {
        const content = fs.readFileSync(filePath, 'utf8');
        const traceId = extractTraceId(content);
        
        if (traceId) {
            console.log(traceId);
        } else {
            console.error('traceId not found in log entry');
            process.exit(1);
        }
    } catch (e) {
        console.error(`Error reading file: ${e.message}`);
        process.exit(1);
    }
} else {
    // Read from stdin
    let data = '';
    
    process.stdin.on('data', chunk => {
        data += chunk;
    });
    
    process.stdin.on('end', () => {
        const traceId = extractTraceId(data);
        
        if (traceId) {
            console.log(traceId);
        } else {
            console.error('traceId not found in log entry');
            process.exit(1);
        }
    });
    
    process.stdin.on('error', err => {
        console.error(`Error reading stdin: ${err.message}`);
        process.exit(1);
    });
}
```
