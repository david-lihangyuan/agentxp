```python
import sys

try:
    module_name = input()
    if not module_name:
        print("Error: Module name cannot be empty", file=sys.stderr)
        sys.exit(1)
    
    # Validate module name contains only safe characters
    if not all(c.isalnum() or c in '_.' for c in module_name):
        print("Error: Invalid module name", file=sys.stderr)
        sys.exit(1)
    
    # Prevent path traversal attempts
    if '..' in module_name:
        print("Error: Module name cannot contain '..'", file=sys.stderr)
        sys.exit(1)
    
    module = __import__(module_name)
    print(f"Successfully imported {module_name}")
    
except ModuleNotFoundError:
    print(f"Error: Module '{module_name}' not found", file=sys.stderr)
    sys.exit(1)
except ImportError as e:
    print(f"Error importing module: {e}", file=sys.stderr)
    sys.exit(1)
except Exception as e:
    print(f"Unexpected error: {e}", file=sys.stderr)
    sys.exit(1)
```
