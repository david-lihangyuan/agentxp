```python
import yaml
import sys
import os

config_file = 'config.yaml'

if not os.path.exists(config_file):
    print(f"Error: {config_file} not found")
    sys.exit(1)

try:
    with open(config_file, 'r') as f:
        config = yaml.safe_load(f)
    
    if not config:
        print("Error: Empty config file")
        sys.exit(1)
    
    if 'services' not in config:
        print("Error: 'services' key not found in config")
        sys.exit(1)
    
    if 'api' not in config['services']:
        print("Error: 'api' key not found in config['services']")
        sys.exit(1)
    
    if 'endpoints' not in config['services']['api']:
        print("Error: 'endpoints' key not found in config['services']['api']")
        sys.exit(1)
    
    endpoints = config['services']['api']['endpoints']
    
    if not isinstance(endpoints, list):
        print("Error: 'endpoints' is not a list")
        sys.exit(1)
    
    if len(endpoints) == 0:
        print("Error: 'endpoints' list is empty")
        sys.exit(1)
    
    print(endpoints[0])
    
except yaml.YAMLError as e:
    print(f"Error parsing YAML: {e}")
    sys.exit(1)
except Exception as e:
    print(f"Error: {e}")
    sys.exit(1)
```
