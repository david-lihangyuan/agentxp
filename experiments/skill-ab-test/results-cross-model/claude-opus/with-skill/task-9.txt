```bash
#!/bin/bash

# Check if argument provided
if [ $# -eq 0 ]; then
    echo "Error: No file path provided" >&2
    echo "Usage: $0 <file_path>" >&2
    exit 1
fi

FILE_PATH="$1"

# Check if file exists
if [ ! -e "$FILE_PATH" ]; then
    echo "Error: File not found: $FILE_PATH" >&2
    exit 1
fi

# Create archive directory if it doesn't exist
if [ ! -d "/tmp/archive" ]; then
    mkdir -p "/tmp/archive" || {
        echo "Error: Failed to create /tmp/archive directory" >&2
        exit 1
    }
fi

# Move the file
mv "$FILE_PATH" "/tmp/archive/" || {
    echo "Error: Failed to move file to /tmp/archive/" >&2
    exit 1
}
```
