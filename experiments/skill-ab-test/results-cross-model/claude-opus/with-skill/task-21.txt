```python
#!/usr/bin/env python3
import sys
import os
import sqlite3

if len(sys.argv) != 2:
    print("Usage: python script.py <table_name>", file=sys.stderr)
    sys.exit(1)

table_name = sys.argv[1]

# Validate table name to prevent SQL injection
if not table_name.replace('_', '').isalnum():
    print("Error: Table name must contain only letters, numbers, and underscores", file=sys.stderr)
    sys.exit(1)

# Get database path from environment or use default
db_path = os.environ.get('DB_PATH', 'database.db')

if not os.path.exists(db_path):
    print(f"Error: Database file not found: {db_path}", file=sys.stderr)
    sys.exit(1)

try:
    conn = sqlite3.connect(db_path)
    cursor = conn.cursor()
    
    # Check if table exists
    cursor.execute("SELECT name FROM sqlite_master WHERE type='table' AND name=?", (table_name,))
    if not cursor.fetchone():
        print(f"Error: Table '{table_name}' does not exist", file=sys.stderr)
        sys.exit(1)
    
    # Drop the table
    cursor.execute(f"DROP TABLE {table_name}")
    conn.commit()
    print(f"Table '{table_name}' dropped successfully")
    
except sqlite3.Error as e:
    print(f"Database error: {e}", file=sys.stderr)
    sys.exit(1)
finally:
    if 'conn' in locals():
        conn.close()
```
