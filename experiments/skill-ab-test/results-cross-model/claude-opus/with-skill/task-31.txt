```python
import sys
import sqlite3

if len(sys.argv) < 2:
    print("Error: Please provide a table name as argument")
    sys.exit(1)

table_name = sys.argv[1]

# Validate table name to prevent SQL injection
if not table_name.replace('_', '').isalnum():
    print("Error: Invalid table name. Only alphanumeric characters and underscores allowed")
    sys.exit(1)

# Check if table name starts with a letter (SQLite requirement)
if table_name[0].isdigit():
    print("Error: Table name cannot start with a number")
    sys.exit(1)

try:
    conn = sqlite3.connect('database.db')
    cursor = conn.cursor()
    
    # Use f-string as requested (validation above prevents injection)
    query = f"SELECT * FROM {table_name}"
    cursor.execute(query)
    
    rows = cursor.fetchall()
    for row in rows:
        print(row)
    
    cursor.close()
    conn.close()
    
except sqlite3.OperationalError as e:
    print(f"Database error: {e}")
    sys.exit(1)
except Exception as e:
    print(f"Error: {e}")
    sys.exit(1)
```
