def normalize(val, min_val, max_val):
    if max_val == min_val:
        raise ValueError("Cannot normalize when max equals min (division by zero)")
    return (val - min_val) / (max_val - min_val)

# Test with max=min
try:
    result = normalize(5, 10, 10)
    print(f"Result: {result}")
except ValueError as e:
    print(f"Error: {e}")
