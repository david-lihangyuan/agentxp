```javascript
#!/usr/bin/env node

let input = '';

process.stdin.setEncoding('utf8');

process.stdin.on('data', chunk => {
  input += chunk;
});

process.stdin.on('end', () => {
  try {
    const result = JSON.parse(input);

    if (
      !result ||
      !result.data ||
      !Array.isArray(result.data.items) ||
      result.data.items.length === 0 ||
      typeof result.data.items[0]?.name !== 'string'
    ) {
      process.exit(1);
    }

    process.stdout.write(result.data.items[0].name + '\n');
  } catch {
    process.exit(1);
  }
});

process.stdin.on('error', () => {
  process.exit(1);
});
```
