```python
#!/usr/bin/env python3
import os
import re
import sys
import sqlite3

def main():
    if len(sys.argv) != 2:
        print(f"Usage: {sys.argv[0]} <table_name>", file=sys.stderr)
        sys.exit(1)

    table_name = sys.argv[1]
    if not re.fullmatch(r"[A-Za-z_][A-Za-z0-9_]*", table_name):
        print("Invalid table name", file=sys.stderr)
        sys.exit(1)

    db_path = os.environ.get("DB_PATH")
    if not db_path:
        print("DB_PATH not set", file=sys.stderr)
        sys.exit(1)
    if not os.path.exists(db_path):
        print(f"Database file not found: {db_path}", file=sys.stderr)
        sys.exit(1)

    try:
        conn = sqlite3.connect(db_path)
        cur = conn.cursor()
        query = f"SELECT * FROM {table_name}"
        cur.execute(query)
        for row in cur.fetchall():
            print(row)
    except sqlite3.Error as e:
        print(f"Database error: {e}", file=sys.stderr)
        sys.exit(1)
    finally:
        try:
            conn.close()
        except Exception:
            pass

if __name__ == "__main__":
    main()
```
