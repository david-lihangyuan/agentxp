const fs = require('fs');
const path = require('path');

module.exports = async function handler(req, res) {
  const requestedPath = req?.query?.path;

  if (typeof requestedPath !== 'string' || requestedPath.trim() === '') {
    res.statusCode = 400;
    res.setHeader('Content-Type', 'application/json');
    res.end(JSON.stringify({ error: 'Missing or invalid "path" query parameter' }));
    return;
  }

  const baseDir = process.cwd();
  const resolvedPath = path.resolve(baseDir, requestedPath);

  if (!resolvedPath.startsWith(baseDir + path.sep) && resolvedPath !== baseDir) {
    res.statusCode = 403;
    res.setHeader('Content-Type', 'application/json');
    res.end(JSON.stringify({ error: 'Access denied' }));
    return;
  }

  let stats;
  try {
    stats = await fs.promises.stat(resolvedPath);
  } catch {
    res.statusCode = 404;
    res.setHeader('Content-Type', 'application/json');
    res.end(JSON.stringify({ error: 'File not found' }));
    return;
  }

  if (!stats.isFile()) {
    res.statusCode = 400;
    res.setHeader('Content-Type', 'application/json');
    res.end(JSON.stringify({ error: 'Path is not a file' }));
    return;
  }

  const stream = fs.createReadStream(resolvedPath);

  stream.on('error', () => {
    if (!res.headersSent) {
      res.statusCode = 500;
      res.setHeader('Content-Type', 'application/json');
      res.end(JSON.stringify({ error: 'Failed to read file' }));
    } else {
      res.destroy();
    }
  });

  res.statusCode = 200;
  res.setHeader('Content-Length', stats.size);
  res.setHeader('Content-Type', 'application/octet-stream');
  res.setHeader('Content-Disposition', `inline; filename="${path.basename(resolvedPath)}"`);

  stream.pipe(res);
};
