```js
const fs = require('fs');
const path = require('path');

const manifestPath = path.resolve(process.cwd(), 'manifest.json');

if (!fs.existsSync(manifestPath)) {
  console.error(`File not found: ${manifestPath}`);
  process.exit(1);
}

let manifest;
try {
  manifest = JSON.parse(fs.readFileSync(manifestPath, 'utf8'));
} catch (err) {
  console.error(`Failed to read or parse manifest.json: ${err.message}`);
  process.exit(1);
}

const scope = manifest?.permissions?.[0]?.scope;

if (scope === undefined) {
  console.error('manifest.permissions[0].scope not found');
  process.exit(1);
}

console.log(scope);
```
