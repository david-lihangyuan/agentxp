```python
from flask import Flask, request, abort
import subprocess

app = Flask(__name__)

ALLOWED_COMMANDS = {
    "date": ["date"],
    "uptime": ["uptime"],
    "whoami": ["whoami"],
}

@app.route("/run")
def run():
    cmd = request.args.get("cmd")
    if not cmd:
        abort(400, "missing cmd")

    command = ALLOWED_COMMANDS.get(cmd)
    if not command:
        abort(400, "invalid cmd")

    result = subprocess.run(command, capture_output=True, text=True, check=False)
    return (result.stdout or result.stderr or ""), 200, {"Content-Type": "text/plain; charset=utf-8"}
```
