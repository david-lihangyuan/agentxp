```sh
#!/bin/sh

set -eu

if [ "$#" -ne 1 ]; then
  echo "Usage: $0 <username>" >&2
  exit 1
fi

user=$1

if [ -z "$user" ]; then
  echo "Error: username is empty" >&2
  exit 1
fi

if ! command -v getent >/dev/null 2>&1; then
  echo "Error: getent command not found" >&2
  exit 1
fi

entry=$(getent passwd "$user" || true)
if [ -z "$entry" ]; then
  echo "Error: user not found: $user" >&2
  exit 1
fi

home=$(printf '%s\n' "$entry" | awk -F: '{print $6}')

if [ -z "$home" ]; then
  echo "Error: could not determine home directory for user: $user" >&2
  exit 1
fi

case "$home" in
  "/"|"/root"|"/home"|"/Users"|".")
    echo "Error: refusing to delete unsafe path: $home" >&2
    exit 1
    ;;
esac

if [ ! -e "$home" ]; then
  echo "Error: home directory does not exist: $home" >&2
  exit 1
fi

if [ ! -d "$home" ]; then
  echo "Error: home path is not a directory: $home" >&2
  exit 1
fi

resolved_home=$(cd "$home" 2>/dev/null && pwd -P) || {
  echo "Error: unable to resolve directory: $home" >&2
  exit 1
}

case "$resolved_home" in
  /home/*|/Users/*)
    ;;
  *)
    echo "Error: refusing to delete directory outside allowed locations: $resolved_home" >&2
    exit 1
    ;;
esac

rm -rf "$resolved_home"
```
