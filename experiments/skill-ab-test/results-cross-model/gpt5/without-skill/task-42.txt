```js
#!/usr/bin/env node

const fs = require('fs');

const input = fs.readFileSync(0, 'utf8').trim();
if (!input) process.exit(1);

let logEntry;
try {
  logEntry = JSON.parse(input);
} catch {
  process.exit(1);
}

let message = logEntry.message ?? logEntry['@message'] ?? logEntry.log ?? logEntry;
if (typeof message === 'string') {
  try {
    message = JSON.parse(message);
  } catch {}
}

const traceId = message?.event?.detail?.metadata?.traceId;
if (traceId !== undefined) process.stdout.write(String(traceId));
```
