function computeRate(amount, duration_seconds) {
  if (duration_seconds === 0) {
    throw new Error('duration_seconds must not be 0');
  }
  return amount / duration_seconds;
}

// Test with duration_seconds = 0
try {
  console.log(computeRate(100, 0));
} catch (err) {
  console.error(err.message);
}
