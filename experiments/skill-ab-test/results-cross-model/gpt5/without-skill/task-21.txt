```python
#!/usr/bin/env python3
import os
import re
import sys
import sqlite3

if len(sys.argv) != 2:
    sys.exit(1)

table = sys.argv[1]

if not re.fullmatch(r"[A-Za-z_][A-Za-z0-9_]*", table):
    sys.exit(1)

db_path = os.environ.get("DB_PATH", "database.db")

conn = sqlite3.connect(db_path)
try:
    conn.execute(f'DROP TABLE "{table}"')
    conn.commit()
finally:
    conn.close()
```
